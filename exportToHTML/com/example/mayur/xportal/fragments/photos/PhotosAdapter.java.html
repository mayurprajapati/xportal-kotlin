240      @Override 
241      public void onBindViewHolder(@NonNull PhotosHolder photosHolder, int pos) { 
242          if (selectedImages.contains(pos)) { 
243              photosHolder.checkBox.setChecked(true); 
244              photosHolder.cardView.setRadius(36); 
245              photosHolder.imageSelectionPlaceholder.setVisibility(View.VISIBLE); 
246          } else { 
247              photosHolder.imageSelectionPlaceholder.setVisibility(View.GONE); 
248              photosHolder.checkBox.setChecked(false); 
249              photosHolder.cardView.setRadius(0); 
250          } 
251   
252          DrawableTransitionOptions options = new DrawableTransitionOptions(); 
253          options.crossFade(); 
254   
255          RequestOptions requestOptions = new RequestOptions().centerCrop().override(150, 150); 
256          Glide.with(mainActivity) 
257                  .load(getUriFromMediaStore(pos)) 
258                  .transition(options) 
259                  .apply(requestOptions) 
260                  .into(photosHolder.mImageView); 
261      }

268      private Cursor swapCursor(Cursor cursor) { 
269          if (mMediaStoreCursor == cursor) { 
270              return null; 
271          } 
272          Cursor oldCursor = mMediaStoreCursor; 
273          this.mMediaStoreCursor = cursor; 
274          if (cursor != null) { 
275              this.notifyDataSetChanged(); 
276          } 
277          return oldCursor; 
278      } 
279   
280      public void changeCursor(Cursor cursor) { 
281          Cursor oldCursor = swapCursor(cursor); 
282          if (oldCursor != null) { 
283              oldCursor.close(); 
284          } 
285      }

287      private Uri getUriFromMediaStore(int position) { 
288          int dataIndex = mMediaStoreCursor.getColumnIndex(MediaStore.Files.FileColumns.DATA); 
289   
290          mMediaStoreCursor.moveToPosition(position); 
291   
292          String dataString = mMediaStoreCursor.getString(dataIndex); 
293          return Uri.parse("file://" + dataString); 
294      }
